    # GRADING — 55 POINTS (v3 Rubric)
    # Cat 1: Performance 20 (Service 10 + Cost 10)
    # Cat 2: S&OP Quality 10
    # Cat 3: Bottleneck & Capacity Decision 10
    # Cat 4: Lean/Quality/Layout Decision 10
    # Cat 5: Justification 5
    # ================================================================
    score_breakdown = {}

    # ------ Category 1: Performance (20 pts) ------
    if service_level >= 98:
        service_points = 10
    elif service_level >= 95:
        service_points = 8
    elif service_level >= 92:
        service_points = 7
    else:
        service_points = 5

    target_costs = {'standard': 65000, 'overtime': 80000, 'two_shift': 100000}
    target_cost = target_costs[capacity_mode]
    cost_ratio = total_cost / target_cost if target_cost > 0 else 1

    if cost_ratio <= 1.05:
        cost_points = 10
    elif cost_ratio <= 1.10:
        cost_points = 8
    elif cost_ratio <= 1.15:
        cost_points = 7
    else:
        cost_points = 5

    score_breakdown['performance'] = service_points + cost_points

    # ------ Category 2: S&OP Quality (10 pts) ------
    total_sop_planned = sum(sop_plan_A) + sum(sop_plan_B)
    expected_demand = (daily_demand_baseline_A + daily_demand_baseline_B) * SIMULATION_DAYS
    sop_ratio = total_sop_planned / expected_demand if expected_demand > 0 else 0

    if 0.95 <= sop_ratio <= 1.05:
        sop_score = 10
    elif 0.90 <= sop_ratio <= 1.10:
        sop_score = 8
    else:
        sop_score = 6

    score_breakdown['sop_quality'] = sop_score

    # ------ Category 3: Bottleneck & Capacity Decision (10 pts) ------
    # Compute true bottleneck from work center utilization
    weekly_totals_A = sop_plan_A  # 8-week weekly totals
    weekly_totals_B = sop_plan_B

    wc_util = {}
    for wc_name, wc in WORK_CENTERS.items():
        total_required_min = sum(
            (weekly_totals_A[i] * wc['sam_a'] + weekly_totals_B[i] * wc['sam_b'])
            for i in range(8)
        )
        total_available_min = wc['capacity_min_day'] * 7 * 8  # all 8 weeks
        wc_util[wc_name] = total_required_min / total_available_min if total_available_min > 0 else 0

    true_bottleneck = max(wc_util, key=lambda k: wc_util[k])
    max_util = wc_util[true_bottleneck]

    student_target_wc = improve_cfg['work_center']  # None if 'none'

    if student_target_wc is None:
        # Student chose no improvement
        if max_util <= 0.90:
            bottleneck_score = 10  # Correct: no critical bottleneck, no improvement needed
        elif max_util <= 1.00:
            bottleneck_score = 6   # Debatable: borderline bottleneck, no action taken
        else:
            bottleneck_score = 2   # Missed critical bottleneck
    else:
        # Student chose to improve a specific work center
        target_util = wc_util.get(student_target_wc, 0)
        if student_target_wc == true_bottleneck:
            # Correct bottleneck identified
            # Check if improvement type is proportionate
            if improve_cfg['multiplier'] >= 1.45 and (max_util * 0.80) <= 1.0:
                # Bought new equipment when modification would have sufficed
                bottleneck_score = 7
            else:
                bottleneck_score = 10
        elif abs(target_util - max_util) / max_util <= 0.10:
            # Close miss — within 10% of true bottleneck
            bottleneck_score = 6
        else:
            bottleneck_score = 3  # Wrong work center

    score_breakdown['bottleneck_score'] = bottleneck_score

    # Store for dispute resolution
    score_breakdown['bottleneck_detail'] = {
        'true_bottleneck': true_bottleneck,
        'true_bottleneck_util': round(max_util * 100, 1),
        'wc_utilizations': {k: round(v * 100, 1) for k, v in wc_util.items()},
        'student_target': student_target_wc,
    }

    # ------ Category 4: Lean / Quality / Layout Decision (10 pts) ------
    layout_pts = 2 if layout_choice == 'functional' else 0
    flow_pts   = 2 if flow_choice == 'cellular' else 0

    # Training: reward choosing training when scrap is material
    if training_choice != 'none':
        training_pts = 3  # Any training chosen
    else:
        # Penalize if service was low (scrap likely contributing)
        training_pts = 1 if service_level >= 95 else 0

    # Lean: reward meaningful quality-at-source choices
    lean_pts_map = {
        'none': 0, '5s': 1,
        'poka_yoke': 3, 'andon': 2,
        'poka_andon_bundle': 3, 'lean_flow': 3
    }
    lean_pts = lean_pts_map.get(lean_choice, 0)

    lean_quality_score = min(10, layout_pts + flow_pts + training_pts + lean_pts)
    score_breakdown['lean_quality_score'] = lean_quality_score

    # ------ Category 5: Justification (5 pts) ------
    if len(justification) >= 500:
        just_score = 5
    elif len(justification) >= 300:
        just_score = 4
    else:
        just_score = 2

    score_breakdown['justification'] = just_score

    # ------ Validity (included in validity flag) ------
    validity_score = 2.0
    validation_flags = []
    weekly_capacity = daily_capacity * 7
    for week_idx in range(8):
        weekly_total = sop_plan_A[week_idx] + sop_plan_B[week_idx]
        if weekly_total > weekly_capacity * 1.3:
            validation_flags.append(f"Week {week_idx+1}: Plan exceeds capacity by >30%")
            validity_score -= 0.5
    validity_score = max(0, validity_score)
    score_breakdown['validity'] = validity_score

    total_score = round(sum([
        score_breakdown['performance'],
        score_breakdown['sop_quality'],
        score_breakdown['bottleneck_score'],
        score_breakdown['lean_quality_score'],
        score_breakdown['justification'],
        score_breakdown['validity']
    ]))

    if total_score >= 50: letter_grade = 'A'
    elif total_score >= 44: letter_grade = 'B'
    elif total_score >= 38: letter_grade = 'C'
    else: letter_grade = 'D'
