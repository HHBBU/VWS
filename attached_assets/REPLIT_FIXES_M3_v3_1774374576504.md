## FIX 4 — Replace Grading Section (Core Change)
**File:** `modules/engine_module3.py`
**Priority:** 🔴 CRITICAL

### ACTION: Find the entire grading block (# GRADING section) and REPLACE with:

```python
    # ================================================================
    # GRADING — 55 POINTS (v3 Rubric)
    # Category 1: Performance 30 pts (Fill 20 + Cost Efficiency 10)
    # Category 2: Inventory Math Correctness 15 pts
    # Category 3: Policy Quality & Reasoning 8 pts (Network 5 + Just 3)
    # Category 4: Validity 2 pts
    # ================================================================
    score_breakdown = {}

    # --------------------------------------------------
    # CATEGORY 1: Performance (30 pts)
    # --------------------------------------------------
    # Fill rate sub-score (0-20)
    if fill_rate >= 94:
        fill_points = 20
    elif fill_rate >= 90:
        fill_points = 17
    elif fill_rate >= 85:
        fill_points = 14
    elif fill_rate >= 80:
        fill_points = 10
    else:
        fill_points = 5

    # Cost efficiency sub-score (0-10)
    target_costs = {
        'centralized': 290000,
        'hybrid':      320000,
        'decentralized': 360000
    }
    target_cost = target_costs.get(network_strategy, 320000)
    cost_ratio = total_cost / target_cost if target_cost > 0 else 1.0

    if cost_ratio <= 1.05:
        cost_points = 10
    elif cost_ratio <= 1.10:
        cost_points = 8
    elif cost_ratio <= 1.15:
        cost_points = 7
    elif cost_ratio <= 1.25:
        cost_points = 5
    else:
        cost_points = 3

    score_breakdown['performance'] = fill_points + cost_points

    # --------------------------------------------------
    # CATEGORY 2: Inventory Math Correctness (15 pts)
    # Compares submitted ROP and Q against formula benchmarks
    # --------------------------------------------------
    avg_daily_demand_actual = total_demand / SIMULATION_DAYS if SIMULATION_DAYS > 0 else 1
    avg_lead_time = (network_config['lead_time_min'] + network_config['lead_time_max']) / 2.0

    # Blended annual holding cost (weighted by M1 forecast proportions)
    total_forecast_for_h = m1_forecast_A + m1_forecast_B
    if total_forecast_for_h > 0:
        blended_H = (
            m1_forecast_A * HOLDING_COST_ANNUAL_A +
            m1_forecast_B * HOLDING_COST_ANNUAL_B
        ) / total_forecast_for_h
    else:
        blended_H = (HOLDING_COST_ANNUAL_A + HOLDING_COST_ANNUAL_B) / 2.0

    # Reference EOQ (engine scale — annual demand from simulation)
    annual_demand_sim = avg_daily_demand_actual * 365
    if blended_H > 0:
        ref_eoq = (2 * annual_demand_sim * ORDERING_COST_S / blended_H) ** 0.5
    else:
        ref_eoq = q  # fallback: no penalty

    # Reference Safety Stock (uses 90% SL = Z=1.28 as minimum standard)
    # σ_assumed: use ~15% of avg daily demand as estimated std dev
    sigma_assumed = avg_daily_demand_actual * 0.15
    ref_ss = 1.28 * sigma_assumed * (avg_lead_time ** 0.5)

    # Reference ROP
    ref_rop = avg_daily_demand_actual * avg_lead_time + ref_ss

    # Score Q against EOQ benchmark
    if ref_eoq > 0:
        q_ratio = q / ref_eoq
        if 0.85 <= q_ratio <= 1.15:          # within ±15%
            q_score = 6
        elif 0.70 <= q_ratio <= 1.30:         # within ±30%
            q_score = 4
        elif 0.50 <= q_ratio <= 1.50:         # within ±50%
            q_score = 2
        else:
            q_score = 0
    else:
        q_score = 3  # fallback partial credit

    # Score ROP against benchmark
    if ref_rop > 0:
        rop_ratio = rop / ref_rop
        if 0.85 <= rop_ratio <= 1.15:
            rop_score = 5
        elif 0.70 <= rop_ratio <= 1.30:
            rop_score = 3
        elif 0.50 <= rop_ratio <= 1.50:
            rop_score = 2
        else:
            rop_score = 0
    else:
        rop_score = 2  # fallback partial credit

    # Safety Stock implicit check
    # Does ROP include a meaningful safety buffer above lead time demand?
    lead_time_demand = avg_daily_demand_actual * avg_lead_time
    student_ss = max(0, rop - lead_time_demand)

    if student_ss >= ref_ss * 0.80:       # SS is at least 80% of reference
        ss_score = 4
    elif student_ss >= ref_ss * 0.40:     # SS is at least 40% of reference
        ss_score = 2
    elif student_ss > 0:                  # SS exists but too small
        ss_score = 1
    else:                                 # No safety stock (ROP = μLT or less)
        ss_score = 0

    math_score = q_score + rop_score + ss_score
    score_breakdown['inventory_logic'] = math_score

    # Store benchmark data for instructor dispute resolution
    score_breakdown['math_benchmark'] = {
        'ref_eoq': round(ref_eoq),
        'ref_rop': round(ref_rop),
        'ref_ss': round(ref_ss),
        'avg_daily_demand': round(avg_daily_demand_actual, 1),
        'avg_lead_time': round(avg_lead_time, 1),
        'q_ratio': round(q / ref_eoq, 3) if ref_eoq > 0 else None,
        'rop_ratio': round(rop / ref_rop, 3) if ref_rop > 0 else None,
        'student_ss': round(student_ss),
        'q_score': q_score,
        'rop_score': rop_score,
        'ss_score': ss_score
    }

    # --------------------------------------------------
    # CATEGORY 3: Policy Quality & Reasoning (8 pts)
    # Network alignment (0-5) + Justification (0-3)
    # --------------------------------------------------
    # Network alignment
    if fill_rate >= 90 and network_strategy in ['hybrid', 'decentralized']:
        net_score = 5
    elif fill_rate >= 90 and network_strategy == 'centralized':
        net_score = 4
    elif fill_rate < 85 and network_strategy == 'centralized':
        net_score = 2
    else:
        net_score = 3

    # Justification quality (character-based)
    if len(justification) >= 400:
        just_score = 3
    elif len(justification) >= 250:
        just_score = 2
    elif len(justification) >= 100:
        just_score = 1
    else:
        just_score = 0

    score_breakdown['network_design'] = net_score
    score_breakdown['justification'] = just_score
    score_breakdown['policy_reasoning'] = net_score + just_score  # combined for display

    # --------------------------------------------------
    # CATEGORY 4: Validity (2 pts)
    # --------------------------------------------------
    validity_score = 2.0
    validation_flags = []

    if rop <= 0 or q <= 0:
        validation_flags.append("ROP or Q cannot be zero or negative")
        validity_score -= 1.0

    if q > 50000:
        validation_flags.append("Order quantity Q exceeds 50,000 — unreasonably high")
        validity_score -= 0.5

    validity_score = max(0.0, validity_score)
    score_breakdown['validity'] = validity_score

    # --------------------------------------------------
    # TOTAL AND LETTER GRADE
    # --------------------------------------------------
    total_score = round(sum([
        score_breakdown.get('performance', 0),
        score_breakdown.get('inventory_logic', 0),
        score_breakdown.get('network_design', 0),
        score_breakdown.get('justification', 0),
        score_breakdown.get('validity', 0)
    ]))

    if total_score >= 50:
        letter_grade = 'A'
    elif total_score >= 44:
        letter_grade = 'B'
    elif total_score >= 38:
        letter_grade = 'C'
    else:
        letter_grade = 'D'
```
